链表

链表是一种重要的线性数据结构，通过指针将节点连接成链状结构。本页面将详细讲解单链表的定义、基本操作和在信息学奥赛中的应用， 帮助NOIP竞赛学生掌握这种灵活的数据结构。

链表概述

什么是链表？

链表是一种线性数据结构，它由一系列节点组成，每个节点包含数据域和指针域。数据域存储数据，指针域存储下一个节点的地址， 通过这种方式将所有节点连接成一个链状结构。

链表与数组的对比：

数组特点：
  • 连续存储，大小固定
  • 随机访问效率高（O(1)）
  • 插入删除效率低（O(n)）
  • 内存利用率低（可能有冗余）
链表特点：
  • 非连续存储，大小动态变化
  • 随机访问效率低（O(n)）
  • 插入删除效率高（O(1)，已知前驱）
  • 内存利用率高（按需分配）

单链表的基本结构

单链表是最简单的链表形式，每个节点包含数据和指向下一个节点的指针：

void deleteNode(int pos) {
  if (head == NULL) return; // 空链表

  Node* temp = head;

    head = temp->next;
    delete temp;
    return;
  }

  for (int i = 1; temp != NULL && i < pos - 1; i++) {

  if (temp == NULL || temp->next == NULL) return;
  
  // 删除节点
  Node* next = temp->next->next;
  delete temp->next;
  temp->next = next;
}

链表基本操作

创建与初始化

#include <iostream>
using namespace std;

// 节点结构体定义
struct Node {
    int data;
    Node* next;
};

// 全局头指针
Node* head = NULL;

// 初始化空链表
void initList() {
    head = NULL;
}

// 创建新节点
Node* createNode(int val) {
    Node* newNode = new Node();
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

尾部插入

// 在链表尾部插入节点
void append(int val) {
    Node* newNode = createNode(val);
    
    // 如果链表为空，直接作为头节点
    if (head == NULL) {
        head = newNode;
        return;
    }
    
    // 找到最后一个节点
    Node* p = head;
    while (p->next != NULL) {
        p = p->next;
    }
    
    // 插入新节点
    p->next = newNode;
}

打印链表

// 打印链表所有元素
void printList() {
    Node* p = head;
    
    // 空链表处理
    if (p == NULL) {
        cout << "链表为空" << endl;
        return;
    }
    
    // 遍历并打印
    cout << "链表元素: ";
    while (p != NULL) {
        cout << p->data << " ";
        p = p->next;
    }
    cout << endl;
}

// 计算链表长度
int getLength() {
    int len = 0;
    Node* p = head;
    while (p != NULL) {
        len++;
        p = p->next;
    }
    return len;
}

查找与修改

// 查找值为val的节点，返回位置(1-based)
int find(int val) {
    int pos = 1;
    Node* p = head;
    
    while (p != NULL) {
        if (p->data == val) {
            return pos;
        }
        p = p->next;
        pos++;
    }
    return -1; // 未找到
}

// 修改第pos个节点的值为val
bool update(int pos, int val) {
    Node* p = head;
    int i = 1;
    
    while (p != NULL && i < pos) {
        p = p->next;
        i++;
    }
    
    if (p == NULL) {
        return false; // 位置无效
    }
    
    p->data = val;
    return true;
}

完整链表实现示例

#include <iostream>
using namespace std;

// 节点结构体
struct Node {
    int data;
    Node* next;
};

Node* head = NULL; // 头指针

// 创建新节点
Node* createNode(int val) {
    Node* newNode = new Node();
    newNode->data = val;
    newNode->next = NULL;
    return newNode;
}

// 在尾部插入
void append(int val) {
    Node* newNode = createNode(val);
    
    if (head == NULL) {
        head = newNode;
        return;
    }
    
    Node* p = head;
    while (p->next != NULL) {
        p = p->next;
    }
    p->next = newNode;
}

// 在指定位置插入
void insertAt(int pos, int val) {
    // 处理插入位置为1的情况
    if (pos == 1) {
        Node* newNode = createNode(val);
        newNode->next = head;
        head = newNode;
        return;
    }
    
    // 找到插入位置的前一个节点
    Node* p = head;
    for (int i = 1; i < pos - 1 && p != NULL; i++) {
        p = p->next;
    }
    
    // 检查位置是否有效
    if (p == NULL) {
        cout << "插入位置无效" << endl;
        return;
    }
    
    // 插入新节点
    Node* newNode = createNode(val);
    newNode->next = p->next;
    p->next = newNode;
}

// 删除指定位置的节点
void deleteAt(int pos) {
    if (head == NULL) {
        cout << "链表为空，无法删除" << endl;
        return;
    }
    
    Node* temp = head;
    
    // 处理删除头节点的情况
    if (pos == 1) {
        head = temp->next;
        delete temp;
        return;
    }
    
    // 找到要删除节点的前一个节点
    for (int i = 1; temp != NULL && i < pos - 1; i++) {
        temp = temp->next;
    }
    
    // 检查位置是否有效
    if (temp == NULL || temp->next == NULL) {
        cout << "删除位置无效" << endl;
        return;
    }
    
    // 删除节点
    Node* next = temp->next->next;
    delete temp->next;
    temp->next = next;
}

// 打印链表
void printList() {
    Node* p = head;
    
    if (p == NULL) {
        cout << "链表为空" << endl;
        return;
    }
    
    cout << "链表元素: ";
    while (p != NULL) {
        cout << p->data << " ";
        p = p->next;
    }
    cout << endl;
}

// 释放链表内存
void freeList() {
    Node* current = head;
    Node* next;
    
    while (current != NULL) {
        next = current->next;
        delete current;
        current = next;
    }
    
    head = NULL;
}

int main() {
    // 测试链表操作
    append(10);
    append(20);
    append(30);
    printList(); // 10 20 30
    
    insertAt(2, 15);
    printList(); // 10 15 20 30
    
    deleteAt(3);
    printList(); // 10 15 30
    
    freeList();
    return 0;
}

链表算法应用

链表反转

链表反转是信奥中常见的操作，通过改变节点的指针方向来实现链表的逆序。 

// 反转链表（迭代法）
void reverseList() {
    Node* prev = NULL;
    Node* current = head;
    Node* next = NULL;
    
    while (current != NULL) {
        next = current->next;  // 保存下一个节点
        current->next = prev;  // 反转当前节点的指针
        prev = current;        // 移动prev到当前节点
        current = next;        // 移动current到下一个节点
    }
    head = prev;  // 更新头指针
}

// 反转链表（递归法）
Node* reverseRecursive(Node* p) {
    // 基准情况
    if (p == NULL || p->next == NULL) {
        return p;
    }
    
    // 递归反转剩余部分
    Node* rest = reverseRecursive(p->next);
    
    // 反转当前节点
    p->next->next = p;
    p->next = NULL;
    
    return rest;
}

// 调用递归反转函数
void reverseListRecursive() {
    head = reverseRecursive(head);
}

链表排序

对链表进行排序时，通常使用归并排序算法，时间复杂度为O(n log n)。 

// 找到链表的中间节点
Node* findMiddle(Node* head) {
    if (head == NULL) return head;
    
    Node* slow = head;
    Node* fast = head->next;
    
    while (fast != NULL && fast->next != NULL) {
        slow = slow->next;
        fast = fast->next->next;
    }
    
    return slow;
}

// 合并两个有序链表
Node* merge(Node* a, Node* b) {
    // 创建临时头节点
    Node* dummy = new Node();
    Node* tail = dummy;
    
    // 合并两个链表
    while (a != NULL && b != NULL) {
        if (a->data <= b->data) {
            tail->next = a;
            a = a->next;
        } else {
            tail->next = b;
            b = b->next;
        }
        tail = tail->next;
    }
    
    // 处理剩余节点
    if (a != NULL) {
        tail->next = a;
    } else {
        tail->next = b;
    }
    
    // 返回合并后的链表
    Node* result = dummy->next;
    delete dummy;
    return result;
}

// 归并排序链表
Node* mergeSort(Node* head) {
    // 基准情况
    if (head == NULL || head->next == NULL) {
        return head;
    }
    
    // 找到中间节点
    Node* mid = findMiddle(head);
    Node* nextOfMid = mid->next;
    
    // 分割链表
    mid->next = NULL;
    
    // 递归排序两部分
    Node* left = mergeSort(head);
    Node* right = mergeSort(nextOfMid);
    
    // 合并排序后的两部分
    return merge(left, right);
}

// 调用排序函数
void sortList() {
    head = mergeSort(head);
}

检测环与环入口

判断链表中是否存在环，并找到环的入口点，是链表的经典问题。 

// 检测链表中是否有环（快慢指针法）
bool hasCycle(Node* head) {
    if (head == NULL || head->next == NULL) {
        return false;
    }
    
    Node* slow = head;
    Node* fast = head->next;
    
    while (slow != fast) {
        if (fast == NULL || fast->next == NULL) {
            return false;
        }
        slow = slow->next;
        fast = fast->next->next;
    }
    
    return true;
}

// 找到环的入口点
Node* findCycleEntry(Node* head) {
    if (head == NULL || head->next == NULL) {
        return NULL;
    }
    
    // 第一步：检测环并找到相遇点
    Node* slow = head;
    Node* fast = head;
    bool hasCycle = false;
    
    while (fast != NULL && fast->next != NULL) {
        slow = slow->next;
        fast = fast->next->next;
        if (slow == fast) {
            hasCycle = true;
            break;
        }
    }
    
    if (!hasCycle) {
        return NULL;
    }
    
    // 第二步：找到环的入口点
    slow = head;
    while (slow != fast) {
        slow = slow->next;
        fast = fast->next;
    }
    
    return slow;
}
