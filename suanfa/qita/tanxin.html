贪心算法

贪心算法是一种在每一步选择中都采取在当前状态下最好或最优的选择，从而希望导致结果是最好或最优的算法。 本页面将详细讲解贪心算法的基本思想、适用场景、实现步骤以及在信息学奥赛中的典型应用。

贪心算法概述

什么是贪心算法？

贪心算法（Greedy Algorithm）是一种在解决问题时，总是做出在当前看来是最好的选择的算法。 它不从整体最优上加以考虑，所做的选择只是在某种意义上的局部最优选择。 虽然贪心算法不能保证总是得到整体最优解，但在许多情况下，它能够产生整体最优解。

贪心算法的核心思想：

  • 分解：将问题分解为若干个子问题
  • 选择：对每个子问题做出局部最优选择
  • 合并：将所有局部最优解合并，得到问题的解

贪心算法的适用条件

贪心算法并非适用于所有问题，它能够得到最优解需要满足两个重要性质：

贪心选择性质

所求问题的整体最优解可以通过一系列局部最优的选择来达到。 即，在做出选择时，不考虑子问题的解，直接做出当前看来最优的选择。

最优子结构性质

问题的最优解包含其子问题的最优解。 如果一个问题的最优解所包含的子问题的解也是最优的，那么该问题具有最优子结构性质。

贪心算法的基本步骤

1
问题分析：分析问题，确定是否可以使用贪心算法解决
2
寻找贪心策略：确定合适的贪心选择标准，即如何做出局部最优选择
3
证明正确性：证明所选择的贪心策略能够得到全局最优解
4
实现算法：根据贪心策略实现算法，通常需要排序等预处理
5
验证与优化：测试算法的正确性，对算法进行优化

活动选择问题是贪心算法的经典应用：有n个活动，每个活动有开始时间和结束时间， 选择最多的活动，使得这些活动互不重叠。贪心策略是：每次选择结束时间最早的活动。

#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
 
// 活动结构体
struct Activity {
  int start;
  int end;
  int id;
};
 
// 排序比较函数
bool compare(Activity a, Activity b) {
  return a.end < b.end;
}
 
// 贪心选择活动
vector<Activity> selectActivities(vector<Activity>& activities) {
  // 按结束时间排序
  sort(activities.begin(), activities.end(), compare);
  
  vector<Activity> result;
  result.push_back(activities[0]);
  
  int lastEnd = activities[0].end;
  
  // 选择后续活动
  for (int i = 1; i < activities.size(); i++) {
    if (activities[i].start >= lastEnd) {
      result.push_back(activities[i]);
      lastEnd = activities[i].end;
    }
  }
  
  return result;
}

1. 硬币找零问题

有不同面额的硬币，用最少的硬币数找出给定金额。贪心策略：每次选择最大面额的硬币。 

#include 
#include 
#include 
using namespace std;

// 贪心算法解决硬币找零问题
int coinChange(vector& coins, int amount) {
    // 排序硬币面额（从大到小）
    sort(coins.rbegin(), coins.rend());
    
    int count = 0;
    int remaining = amount;
    
    // 贪心选择：每次取最大面额的硬币
    for (int coin : coins) {
        if (remaining >= coin) {
            int num = remaining / coin;
            count += num;
            remaining -= num * coin;
            cout << "使用 " << num << " 个 " << coin << " 元硬币" << endl;
        }
        
        // 如果已经找零完成，退出循环
        if (remaining == 0) {
            break;
        }
    }
    
    // 如果无法找零（remaining不为0），返回-1
    return remaining == 0 ? count : -1;
}

int main() {
    // 硬币面额（必须包含1元，否则可能无法找零）
    vector coins = {1, 5, 10, 20, 50, 100};
    int amount;
    
    cout << "请输入要找零的金额: ";
    cin >> amount;
    
    int result = coinChange(coins, amount);
    
    if (result != -1) {
        cout << "最少需要 " << result << " 个硬币" << endl;
    } else {
        cout << "无法完成找零" << endl;
    }
    
    return 0;
}

2. 区间调度问题

选择最多的非重叠区间。贪心策略：每次选择结束时间最早的区间。 

#include 
#include 
#include 
using namespace std;

// 区间结构体
struct Interval {
    int start;  // 开始时间
    int end;    // 结束时间
    int id;     // 区间编号
    
    Interval(int s, int e, int i) : start(s), end(e), id(i) {}
};

// 排序比较函数：按结束时间从小到大排序
bool compareIntervals(const Interval& a, const Interval& b) {
    return a.end < b.end;
}

// 贪心算法选择最多的非重叠区间
vector selectIntervals(vector& intervals) {
    // 按结束时间排序
    sort(intervals.begin(), intervals.end(), compareIntervals);
    
    vector selected;
    if (intervals.empty()) return selected;
    
    // 选择第一个区间
    selected.push_back(intervals[0]);
    int lastEnd = intervals[0].end;
    
    // 依次选择后续区间
    for (size_t i = 1; i < intervals.size(); ++i) {
        // 如果当前区间的开始时间 >= 最后选中区间的结束时间
        if (intervals[i].start >= lastEnd) {
            selected.push_back(intervals[i]);
            lastEnd = intervals[i].end;
        }
    }
    
    return selected;
}

int main() {
    // 创建一些区间
    vector intervals;
    intervals.emplace_back(1, 4, 1);
    intervals.emplace_back(3, 5, 2);
    intervals.emplace_back(0, 6, 3);
    intervals.emplace_back(5, 7, 4);
    intervals.emplace_back(3, 9, 5);
    intervals.emplace_back(5, 9, 6);
    intervals.emplace_back(6, 10, 7);
    intervals.emplace_back(8, 11, 8);
    intervals.emplace_back(8, 12, 9);
    intervals.emplace_back(2, 14, 10);
    intervals.emplace_back(12, 16, 11);
    
    // 选择最多的非重叠区间
    vector result = selectIntervals(intervals);
    
    // 输出结果
    cout << "最多可以选择 " << result.size() << " 个非重叠区间：" << endl;
    for (const auto& interval : result) {
        cout << "区间 " << interval.id << "：[" << interval.start << ", " << interval.end << ")" << endl;
    }
    
    return 0;
}

3. 哈夫曼编码

哈夫曼编码是一种用于数据压缩的贪心算法。贪心策略：每次选择两个频率最低的节点合并。 

#include 
#include 
#include 
#include 
#include 
using namespace std;

// 哈夫曼树节点
struct HuffmanNode {
    char data;           // 字符
    int freq;            // 频率
    HuffmanNode *left, *right; // 左右子树
    
    HuffmanNode(char d, int f) : data(d), freq(f), left(nullptr), right(nullptr) {}
};

// 比较函数，用于优先队列（最小堆）
struct CompareNodes {
    bool operator()(HuffmanNode* a, HuffmanNode* b) {
        return a->freq > b->freq; // 频率小的节点优先级高
    }
};

// 构建哈夫曼树
HuffmanNode* buildHuffmanTree(unordered_map& freq) {
    // 创建最小堆
    priority_queue, CompareNodes> minHeap;
    
    // 将所有字符作为叶子节点加入堆
    for (auto& pair : freq) {
        minHeap.push(new HuffmanNode(pair.first, pair.second));
    }
    
    // 合并节点直到堆中只剩一个节点
    while (minHeap.size() > 1) {
        // 取出两个频率最小的节点
        HuffmanNode* left = minHeap.top();
        minHeap.pop();
        
        HuffmanNode* right = minHeap.top();
        minHeap.pop();
        
        // 创建新的内部节点，频率为两个节点频率之和
        HuffmanNode* internal = new HuffmanNode('$', left->freq + right->freq);
        internal->left = left;
        internal->right = right;
        
        // 将新节点加入堆
        minHeap.push(internal);
    }
    
    // 返回根节点
    return minHeap.top();
}

// 生成哈夫曼编码
void generateCodes(HuffmanNode* root, string code, unordered_map& codes) {
    if (root == nullptr) return;
    
    // 如果是叶子节点，保存编码
    if (root->data != '$') {
        codes[root->data] = code;
        return;
    }
    
    // 左子树加0，右子树加1
    generateCodes(root->left, code + "0", codes);
    generateCodes(root->right, code + "1", codes);
}

// 哈夫曼编码主函数
unordered_map huffmanEncoding(unordered_map& freq) {
    // 构建哈夫曼树
    HuffmanNode* root = buildHuffmanTree(freq);
    
    // 生成编码
    unordered_map codes;
    generateCodes(root, "", codes);
    
    return codes;
}

int main() {
    // 字符频率示例
    unordered_map freq = {
        {'a', 5}, {'b', 9}, {'c', 12}, {'d', 13}, {'e', 16}, {'f', 45}
    };
    
    // 获取哈夫曼编码
    unordered_map codes = huffmanEncoding(freq);
    
    // 输出结果
    cout << "哈夫曼编码结果：" << endl;
    for (auto& pair : codes) {
        cout << pair.first << " : " << pair.second << endl;
    }
    
    return 0;
}

贪心算法的奥赛应用

信息学奥赛中常见的贪心问题类型

区间相关问题

包括区间调度、区间覆盖、区间选点等问题，通常通过排序区间的起点或终点来贪心选择。

资源分配问题

如任务分配、机器调度等，通常根据某种优先级（如耗时、利润）进行排序后分配。

排序相关问题

通过自定义排序规则解决的问题，如拼接最大数、最优排列等。

背包变种问题

分数背包问题可以用贪心算法解决，通过计算单位重量价值排序选择。

Huffman编码类问题

基于频率的最优编码问题，通过优先队列选择合并两个最小频率元素。

图论中的贪心问题

如最小生成树（Kruskal和Prim算法）、最短路径（Dijkstra算法）等。

贪心算法的解题技巧

  • 找到合适的贪心策略：这是解决贪心问题的关键，通常需要对问题进行深入分析
  • 排序是常用手段：大多数贪心问题需要先对数据进行排序，排序依据就是贪心策略的体现
  • 证明策略的正确性：可以尝试使用反证法，证明如果不采用该策略，结果不会更优
  • 从简单案例入手：通过小规模案例找到规律，归纳出贪心策略
  • 注意边界情况：如空输入、只有一个元素等特殊情况的处理
  • 结合其他算法：在复杂问题中，贪心算法可能作为整体算法的一部分使用
